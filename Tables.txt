CREATE TABLE players (
    player_id INT NOT NULL, 
    player_name VARCHAR, 
    experience INT,   
    rank_id INT NOT NULL,                 
    team_id INT,                           
         
);

CREATE TABLE Team(

	Team_id INT NOT NULL,
	Team_name VARCHAR,
	Team_Creation DATE,
	Region VARCHAR(50)

);

CREATE TABLE Match(

	Match_id INT NOT NULL,
	Match_Date DATE,
	Duration INTERVAL,
	Winning_team_id INT,
	Losing_team_id INT

);

CREATE TABLE Team_PlayerTable(
Team_ID INT NOT NULL
Player_ID INT NOT NULL
);

CREATE TABLE Game_StatsTable(
Player_ID INT NOT NULL
Match_id INT NOT NULL
Kills INT 
Deaths INT 
Assists INT
);

CREATE TABLE RankTable(
Rank_ID INT PRIMARY KEY
Rank_Name VARCHAR
Min_Experience INT
Max_Experience INT
);
