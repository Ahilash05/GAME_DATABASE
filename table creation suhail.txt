CREATE TABLE players (
    player_id INT NOT NULL, 
    player_name VARCHAR, 
    experience INT,   
    rank_id INT NOT NULL,                 
    team_id INT,                           
         
);

CREATE TABLE Team(

	Team_id INT NOT NULL,
	Team_name VARCHAR,
	Team_Creation DATE,
	Region VARCHAR(50)

);

CREATE TABLE Match(

	Match_id INT NOT NULL,
	Match_Date DATE,
	Duration INTERVAL,
	Winning_team_id INT,
	Losing_team_id INT

);
